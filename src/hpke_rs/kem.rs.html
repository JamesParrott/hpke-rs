15pub(crate) fn encaps<Crypto: HpkeCrypto>(
16    hpke: &mut Hpke<Crypto>,
17    pk_r: &[u8],
18) -> Result<(Vec<u8>, Vec<u8>), Error> {
19    let alg = hpke.kem_id;
20    match alg {
21        KemAlgorithm::DhKemP256
22        | KemAlgorithm::DhKemK256
23        | KemAlgorithm::DhKemP384
24        | KemAlgorithm::DhKemP521
25        | KemAlgorithm::DhKem25519
26        | KemAlgorithm::DhKem448 => {
27            let randomness = hpke
28                .random(alg.private_key_len())
29                .map_err(|_| Error::InsufficientRandomness)?;
30            dh_kem::encaps::<Crypto>(alg, pk_r, &ciphersuite(alg), &randomness)
31        }
32        KemAlgorithm::XWingDraft06 => Crypto::kem_encaps(alg, pk_r, hpke.rng()),
33    }
34}

93/// Returns (private, public)
94pub(crate) fn key_gen<Crypto: HpkeCrypto>(
95    alg: KemAlgorithm,
96    prng: &mut Crypto::HpkePrng,
97) -> Result<(Vec<u8>, Vec<u8>), Error> {
98    match alg {
99        // For ECDH based keys, we generate a completely fresh key.
100        KemAlgorithm::DhKemP256
101        | KemAlgorithm::DhKemK256
102        | KemAlgorithm::DhKemP384
103        | KemAlgorithm::DhKemP521
104        | KemAlgorithm::DhKem25519
105        | KemAlgorithm::DhKem448 => dh_kem::key_gen::<Crypto>(alg, prng),
106        KemAlgorithm::XWingDraft06 => {
107            // For XWing we use the derive key pair function.
108            let mut seed = vec![0u8; alg.private_key_len()];
109            prng.fill_bytes(&mut seed);
110            let (pk, sk) = derive_key_pair::<Crypto>(alg, &seed)?;
111            Ok((sk, pk))
112        }
113    }
114}

116/// Derive key pair from the input key material `ikm`.
117///
118/// Returns (PublicKey, PrivateKey).
119pub(crate) fn derive_key_pair<Crypto: HpkeCrypto>(
120    alg: KemAlgorithm,
121    ikm: &[u8],
122) -> Result<(PublicKey, PrivateKey), Error> {
123    match alg {
124        KemAlgorithm::DhKemP256
125        | KemAlgorithm::DhKemK256
126        | KemAlgorithm::DhKemP384
127        | KemAlgorithm::DhKemP521
128        | KemAlgorithm::DhKem25519
129        | KemAlgorithm::DhKem448 => dh_kem::derive_key_pair::<Crypto>(alg, &ciphersuite(alg), ikm),
130        KemAlgorithm::XWingDraft06 => {
131            let seed = libcrux_sha3::shake256::<32>(ikm);
132            let kp = Crypto::kem_key_gen_derand(alg, &seed)?;
133            Ok(kp)
134        }
135    }
